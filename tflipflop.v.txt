module t_ff(
input wire clk,
input  rst,
input t,
output reg q
);

 always @(posedge clk or posedge rst) begin//sensitivity

       if(rst) begin 
              q<= 1'b0;
       end

       else begin
              case(t)
                   1'b0: q<=q;
                   1'b1: q<=~q;
              endcase
        end
end
endmodule

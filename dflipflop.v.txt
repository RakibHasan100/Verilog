module d_ff(
input wire clk,
input  rst,
input d,
output reg q
);

 always @(posedge clk or posedge rst) begin//sensitivity

       if(rst) begin 
              q<= 1'b0;
       end

       else begin
              case(d)
                   1'b0: q<=1'b0;//q<=d;
                   1'b1: q<=1'b1;//q<=d;
              endcase
        end
end
endmodule
